# 1) Idea Scout — research sprint, daily at 02:00 UTC (6 PM PST)
0 2 * * * cd /home/n8garvie/.openclaw/workspace/mission-control && bash scripts/heartbeat-scout-ideas.sh >> logs/scout-cron.log 2>&1

# 2) Build Monitor — every 2 minutes. Picks up approved ideas, drives the
#    forge → pixel → echo → integrator → deploy pipeline, escalates stuck builds.
#    The script grabs an exclusive flock at startup; concurrent firings exit
#    immediately so a 2-minute cadence is safe.
*/2 * * * * cd /home/n8garvie/.openclaw/workspace/mission-control && /home/n8garvie/.nvm/versions/node/v22.22.0/bin/node scripts/build-monitor.js >> logs/build-monitor.log 2>&1

# 3) Overnight build sweep — daily at 06:00 UTC (10 PM PST). Belt-and-braces
#    fallback to clear any backlog that the every-2-min monitor missed (e.g. if
#    cron was paused).
0 6 * * * cd /home/n8garvie/.openclaw/workspace/mission-control && bash scripts/overnight-build.sh >> logs/build-cron.log 2>&1
